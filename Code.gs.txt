function doPost(e) {
  const data = JSON.parse(e.posrData.contents);


  if (data.type == "photo"){
    const imageData = data.image;

    const base64Data = imageData.split(",")[1];

    const blob = Utilities.newBlob(
      Utilities.base64Decode(base64Data),
      "image/png",
      Date().toString() + ".png"
    );

    const FOLDER_ID ="1FLZJ7hZM1SGHOnCfoDk18waJ2CVPvukX";
    const folder = DriveApp.getFolderById(FOLDER_ID);
    const file = folder.createFile(blob)
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);

    const fileId = file.getId();
    const imageUrl = "=IMAGE(\"https://drive.google.com/uc?export=view&id=" + fileId + "\")";

    const fichier = data.fichier;
    const feuille = data.feuille;

    const sheet = SpreadsheetApp
      .openById(fichier)
      .getSheetByName(feuille);

    sheet.appendRow([imageUrl]);

    return ContentService.createTextOutput(fichier);
  } 


  if (data.type == "MAJfichier"){
      const props = PropertiesService.getScriptProperties();
      props.setProperty("fichier", data.nom);
      return ContentService.createTextOutput(data.nom);
  }


  if (data.type == "MAJfeuille"){
      const props = PropertiesService.getScriptProperties();
      props.setProperty("feuille", data.nom);
      return ContentService.createTextOutput(data.nom);
  }


  if (data.type == "upload") {
    const props = PropertiesService.getScriptProperties();

    const key = data.key;

    const value = JSON.stringify(data.content);

    props.setProperty(key, value);

    return ContentService.createTextOutput(
      JSON.stringify({ ok: true })
    ).setMimeType(ContentService.MimeType.JSON);
  }


  if (data.type == "download") {
    const props = PropertiesService.getScriptProperties();

    const key = data.key;

    const value = props.getProperty(key);

    return ContentService.createTextOutput(value || "null")
      .setMimeType(ContentService.MimeType.JSON);
  }

}
